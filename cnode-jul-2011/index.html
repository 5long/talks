Roadmap

0.2: Plugin System

Declaring Dependencies

var File = require('roil/resource/file')

exports.stylus = function(roil) {
  var endWithCss = /\.css$/

  roil.matchResource(File, endWithCss, function(r) {
    var path = r.fullPath
      , stylus = path.replace(endWithCss, ".styl")
    r.addDep(stylus)
  })
}

Mount Middleware

var httpProxy = require("http-proxy")

exports.rails = function(roil) {
  roil.server.use(
    httpProxy.createServer(3000, 'localhost')
  )
}

Validation

var File = require('roil/resource/file')
  , jsLint = require('jslint')

exports.jsLint = function(roil) {
  roil.matchResource(File, function(r) {
    r.on('needValidation', function(e) {
      e.pause(2000)
      jsLint.test(r.fullPath, function(err) {
        if (err) {
          e.stop()
          console.error("%s failed js lint test", r)
        } else {
          e.proceed()
        }
      })
    })
  })
}
